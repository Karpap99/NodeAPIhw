Документація по роутам
  <id> - обов'язково Number
GET:
/students // Запит до всіх студентів
  body{ 
    "SortBy":"rate || group" // параметр сортування, якщо не потрібен то робимо запит без body
  }
/student/<id> // Запит до конкретного студента
PUT:
/student/<id> // Зміна інформації конкретного студента
  body{  // Якщо поле не потребуе зміни, просто не додаємо його в body
    "firstName": "...", 
    "lastName": "...",
    "group": "...",
    "rate": "..."
  }
POST:
/student/<id> // Додавання студента в базу данних
  body{  
    "firstName": "...", // Обов'язкове поле, не може бути пусте
    "lastName": "...", // Обов'язкове поле, не може бути пусте
    "group": "...", // Не обов'язкове, за замовчуванням 0
    "rate": "..." // Не обов'язкове, за замовчуванням 0
  }
DELETE:
/student/<id> // Видалення студенту з бази данних
